SELECT
    -- Dados do Artista (artists.parquet)
    art.name AS artist_name,
    art.followers_total,
    art.popularity AS artist_popularity,
    
    -- Gêneros do Artista (artist_genres.parquet)
    agen.genre,
    
    -- Imagens do Artista (artist_images.parquet)
    aimg.url AS artist_image_url,
    aimg.width AS artist_image_width,
    aimg.height AS artist_image_height,

    -- Dados do Álbum (albums.parquet)
    alb.name AS album_name,
    alb.album_type,
    alb.release_date,
    alb.label,
    alb.popularity AS album_popularity,
    alb.total_tracks,
    
    -- Imagens do Álbum (album_images.parquet)
    alimg.url AS album_image_url,
    alimg.width AS album_image_width,
    alimg.height AS album_image_height,

    -- Dados da Faixa (tracks.parquet)
    t.name AS track_name,
    t.track_number,
    t.duration_ms,
    t.popularity AS track_popularity,
    t.explicit,
    t.preview_url,
    
    -- Mercados Disponíveis (available_markets.parquet)
    am.available_markets

FROM tracks t
-- Relaciona faixa com álbum
LEFT JOIN albums alb ON t.album_rowid = alb.rowid [cite: 66, 15]
-- Relaciona álbum com mercados
LEFT JOIN available_markets am ON alb.available_markets_rowid = am.rowid [cite: 15, 52]
-- Relaciona imagens do álbum
LEFT JOIN album_images alimg ON alb.rowid = alimg.album_rowid [cite: 15, 8]
-- Relaciona faixas com artistas (tabela associativa n:n)
LEFT JOIN track_artists ta ON t.rowid = ta.track_rowid [cite: 66, 59]
-- Relaciona artistas
LEFT JOIN artists art ON ta.artist_rowid = art.rowid [cite: 59, 45]
-- Relaciona gêneros dos artistas
LEFT JOIN artist_genres agen ON art.rowid = agen.artist_rowid [cite: 45, 31]
-- Relaciona imagens dos artistas
LEFT JOIN artist_images aimg ON art.rowid = aimg.artist_rowid [cite: 45, 37]
